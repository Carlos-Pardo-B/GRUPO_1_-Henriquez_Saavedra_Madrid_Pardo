import express from 'express';
import cors from 'cors';
import authRoutes from './routes/authRoutes.js';
import publicRoutes from './routes/publicRoutes.js';
import adminRoutes from './routes/adminRoutes.js';
import orgRoutes from './routes/orgRoutes.js';
import { authRequired } from './middleware/authMiddleware.js';
import { superAdminRequired } from './middleware/adminMiddleware.js';

const app = express();

// Middlewares
app.use(express.json());

// Configuración de CORS: permitir cualquier origen
const corsOptions = {
  origin: (origin, callback) => callback(null, true), // refleja el origen que haga la petición
  credentials: true,
  methods: ['GET', 'POST', 'PUT', 'PATCH', 'DELETE', 'OPTIONS'],
  allowedHeaders: ['Content-Type', 'Authorization', 'Accept']
};

app.use(cors(corsOptions));

// Rutas
app.get('/', (req, res) => {
  res.json({ 
    message: 'MemorialConnect API is running',
    version: '1.0.0',
    endpoints: {
      auth: '/auth',
      docs: '/api-docs' // Si tienes Swagger
    }
  });
});

// Rutas de autenticación
app.use('/auth', authRoutes);

// Rutas públicas
\r\n// Rutas de organización (requiere auth)\r\napp.use('/org', authRequired, orgRoutes);\r\n
// Rutas de administración (solo superadmin)
app.use('/admin', authRequired, superAdminRequired, adminRoutes);

// Middleware para manejar rutas no encontradas
app.use((req, res, next) => {
  res.status(404).json({
    error: 'Route not found',
    path: req.path,
    method: req.method
  });
});

// Middleware para manejar errores globales
app.use((err, req, res, next) => {
  console.error('Error:', err.stack);
  
  const statusCode = err.status || 500;
  const message = err.message || 'Internal server error';
  
  res.status(statusCode).json({
    error: message,
    ...(process.env.NODE_ENV === 'development' && { stack: err.stack })
  });
});

export default app;
